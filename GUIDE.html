ğŸ”‘ Step 2 â€” Role-Based Authentication

The system has two roles: user (student) and admin. Role is saved in Firestore when signing up.

How Roles Work

1
User signs up â†’ auth.js creates Firebase Auth account â†’ saves to users collection with role: "user"
2
If email matches admin emails list â†’ role: "admin" is assigned automatically
3
On login â†’ role is read from Firestore â†’ redirect to correct dashboard
4
Each dashboard calls protectPage("user") or protectPage("admin") â†’ wrong role = redirect

How to Make Someone Admin

Option 1: Add their email to the ADMIN_EMAILS array in auth.js before they sign up.

Option 2: In Firestore Console â†’ users collection â†’ find their document â†’ change role field to "admin" manually.

âš¡ Step 7 â€” Real-Time Updates

Instead of getDocs() (one-time fetch), we use onSnapshot() which fires immediately and again every time data changes in Firestore.

Simple Example

// onSnapshot = "keep watching this collection"
onSnapshot(query(collection(db, "complaints"), where("userId", "==", uid)), (snapshot) => {
  // This runs when page loads AND every time data changes
  const complaints = snapshot.docs.map(d => ({ id: d.id, ...d.data() }));
  renderTable(complaints); // Automatically updates the UI
});

Result: When admin changes a status â†’ user sees the new badge instantly â€” no page refresh needed!

ğŸ”’ Step 8 â€” Firestore Security Rules

Go to Firestore â†’ Rules tab â†’ paste these rules:

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // Helper: check if request is from authenticated user
    function isLoggedIn() {
      return request.auth != null;
    }

    // Helper: check if user is admin
    function isAdmin() {
      return isLoggedIn() &&
        get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == "admin";
    }

    // Helper: check if user owns this document
    function isOwner(userId) {
      return isLoggedIn() && request.auth.uid == userId;
    }

    // USERS collection
    match /users/{userId} {
      allow read: if isLoggedIn();
      allow write: if isOwner(userId) || isAdmin();
    }

    // LOST & FOUND
    match /lost_found_items/{docId} {
      allow read: if isLoggedIn();
      allow create: if isLoggedIn();
      allow update: if isAdmin() || isOwner(resource.data.userId);
      allow delete: if isAdmin();
    }

    // COMPLAINTS
    match /complaints/{docId} {
      allow read: if isLoggedIn();
      allow create: if isLoggedIn();
      allow update: if isAdmin();  // Only admin updates status
      allow delete: if isAdmin();
    }

    // VOLUNTEERS
    match /volunteers/{docId} {
      allow read: if isLoggedIn();
      allow create: if isLoggedIn();
      allow update, delete: if isAdmin();
    }

    // NOTIFICATIONS
    match /notifications/{docId} {
      allow read: if isLoggedIn() && resource.data.userId == request.auth.uid;
      allow create: if isLoggedIn();
      allow update: if isLoggedIn() && resource.data.userId == request.auth.uid;
      allow delete: if isAdmin();
    }
  }
}
